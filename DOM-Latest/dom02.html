script.js // Wait until the DOM is ready document.addEventListener("DOMContentLoaded", () => { const resultDiv = document.querySelector("#result"); const calculateBtn = document.querySelector("#calculateBtn"); const resetBtn = document.querySelector("#resetBtn"); // ‚úÖ Calculate button event calculateBtn.addEventListener("click", () => { const sno = getValue("sno"); const sname = getValue("sname"); const sclass = getValue("sclass"); const section = getValue("section"); const marks = Array.from({ length: 6 }, (_, i) => parseFloat(getValue(`mark${i + 1}`)) ); // Validation if (!sno || !sname || !sclass || !section || marks.some(isNaN)) { resultDiv.innerHTML = `

‚ö†Ô∏è Please fill in all fields correctly.

`; return; } const { total, average, grade, passOrFail } = calculateGrade(marks); resultDiv.innerHTML = `

Student Number: ${sno}

Student Name: ${sname}

Class: ${sclass}

Section: ${section}

Total Marks: ${total}

Average Marks: ${average.toFixed(2)}

Grade: ${grade}

Result: ${passOrFail}

`; }); // ‚úÖ Reset button event resetBtn.addEventListener("click", () => { document.querySelectorAll("input").forEach(input => (input.value = "")); resultDiv.innerHTML = ""; }); }); // üîπ Helper function to get input value by id const getValue = id => document.getElementById(id).value.trim(); // üîπ Function to calculate total, average, and grade const calculateGrade = marks => { const total = marks.reduce((sum, mark) => sum + mark, 0); const average = total / marks.length; const isFail = marks.some(mark => mark < 35); let grade; if (isFail) grade = "F"; else if (average >= 90) grade = "A"; else if (average >= 75) grade = "B"; else if (average >= 60) grade = "C"; else if (average >= 50) grade = "D"; else grade = "E"; const passOrFail = isFail ? "Fail" : "Pass"; return { total, average, grade, passOrFail }; };
